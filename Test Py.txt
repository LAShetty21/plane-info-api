url = "https://api.adsb.lol/v2/closest/48.3158156/11.7105641/1"

response = requests.get(url)
if response.status_code == 200:
    data = json.loads(response.text)
    
    aircraft_list = data.get("ac", [])

    if not aircraft_list:  # If list is empty or None
        print("No flights found")
    else:


        for ac in aircraft_list:
            callsign = ac.get("flight") or ac.get("callsign") or "Unknown"
            aircraft_type = ac.get("t") or ac.get("type") or "Unknown"

            # Extract ICAO airline code (first 3 letters of callsign)
            airline_code = callsign[:3].upper()
            airline_name = AIRLINE_CODES.get(airline_code, "Unknown Airline")
        
            print(f"Airline: {airline_name}, Aircraft Type: {aircraft_type}")
else:
    print(f"Error fetching data: {response.status_code}")
